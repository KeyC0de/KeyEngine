153template<typename T>
154T* alignForward( const T *p,
155 const std::size_t alignment ) noexcept
156{
157 if ( alignment == 0 )
158 {
159 return p;
160 }
161 const std::uintptr_t ip = reinterpret_cast<std::uintptr_t>( p );
162 if ( ip % alignment == 0 )
163 {
164 return p;
165 }
166 return reinterpret_cast<T*>( ( ip + ( alignment - 1 ) ) & ~( alignment - 1 ) );
167 // or: (ip + alignment - 1) / alignment * alignment;
168}

423template<typename TContainer, typename TComparator>
424bool hasDuplicates( const TContainer &container,
425 const TComparator &comparator )
426{
427 using TIt = typename TContainer::const_iterator;
428 using T = typename TContainer::value_type;
429
430 bool is_unique = true;
431
432 for ( TIt end = std::end( container ), itr1 = std::begin( container ); is_unique && itr1 != end; itr1 = std::next( itr1 ) )
433 {
434 //If the type contained in the container is a pointer or a number we pass everything by value
435 if constexpr ( std::is_pointer_v<T> || std::is_arithmetic_v<T> )
436 {
437 const auto uniqueness_predicate = [&comparator, lhs = *itr1]( const T rhs )
438 {
439 return comparator( lhs, rhs );
440 };
441
442 is_unique = std::none_of( std::next( itr1 ), end, uniqueness_predicate );
443 }
444 // Otherwise pass everything by const lval ref
445 else
446 {
447 const auto uniqueness_predicate = [&comparator, &lhs = *itr1]( typename TContainer::const_reference rhs )
448 {
449 return comparator( lhs, rhs );
450 };
451
452 is_unique = std::none_of( std::next( itr1 ), end, uniqueness_predicate );
453 }
454 }
455
456 return is_unique == false;
457}
