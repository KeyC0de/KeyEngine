170 template<template<VertexInputLayout::ILEementType> class T, typename... Args>
171 static constexpr auto bridge( const VertexInputLayout::ILEementType type,
172 Args&&... args ) cond_noex
173 {
174 switch( type )
175 {
176#define X( el ) case VertexInputLayout::el: \
177 { \
178 return T<VertexInputLayout::el>::exec( std::forward<Args>( args )... ); \
179 }
180 VERTEX_INPUT_LAYOUT_ELEMENT_TYPES
181#undef X
182 }
183 // default:
184 ASSERT( false, "Invalid element type query." );
