289// protecting against the the case where the old range is 0 ( oldMin = oldMax )
290template<typename T, typename = std::enable_if_t<std::is_arithmetic_v<T>>>
291constexpr T convertRange( const T valueInRange,
292 const T oldMin,
293 const T oldMax,
294 const T newMin,
295 const T newMax )
296{
297 T oldRange = oldMax - oldMin;
298 T newValueInRange;
299 if ( oldRange == 0 )
300 {
301 newValueInRange = newMin;
302 }
303 else
304 {
